const edges = [

  ['n-qasim','n-hashim'],['n-qasim','n-dan'],['n-qasim','n-abdul'],

  ['n-mohsin','n-marwa'],['n-mohsin','n-daniel'],

  ['n-joe','n-joeel'],['n-joe','n-chris'],['n-joe','n-preston'],

  ['n-joe','n-scott'],['n-joe','n-jenna'],['n-joe','n-jamie'],['n-joe','n-justin'],

  ['n-thomas','n-clixio'],['n-thomas','n-graphic'],

];


const dottedExtra = [['n-andro','n-dan']];


function draw() {

  const chart = document.getElementById('chart');

  const svg   = document.getElementById('lines');

  const cr    = chart.getBoundingClientRect();


  chart.style.minHeight = chart.scrollHeight + 'px';

  svg.setAttribute('viewBox', `0 0 ${cr.width} ${chart.scrollHeight}`);

  svg.innerHTML = '';


  function mid(id) {

    const el = document.getElementById(id);

    if (!el) return null;

    const r = el.getBoundingClientRect();

    return {

      cx:  r.left - cr.left + r.width / 2,

      top: r.top  - cr.top,

      bot: r.top  - cr.top + r.height,

    };

  }


  const byParent = {};

  for (const [p, c] of edges) {

    if (!byParent[p]) byParent[p] = [];

    byParent[p].push(c);

  }


  const DASH = '5 4';

  const COL  = '#2e78cc';

  const W    = 1.6;

  const R    = 5;


  function line(d) {

    const el = document.createElementNS('http://www.w3.org/2000/svg', 'path');

    el.setAttribute('d', d);

    el.setAttribute('stroke', COL);

    el.setAttribute('stroke-width', W);

    el.setAttribute('stroke-dasharray', DASH);

    el.setAttribute('fill', 'none');

    el.setAttribute('stroke-linecap', 'round');

    svg.appendChild(el);

  }


  for (const [parentId, childIds] of Object.entries(byParent)) {

    const parent = mid(parentId);

    if (!parent) continue;

    const children = childIds.map(mid).filter(Boolean);

    if (!children.length) continue;


    const childTopY = Math.min(...children.map(c => c.top));

    const midY = parent.bot + (childTopY - parent.bot) / 2;


    if (children.length === 1) {

      const c = children[0];

      if (Math.abs(parent.cx - c.cx) < 2) {

        line(`M ${parent.cx} ${parent.bot} L ${c.cx} ${c.top}`);

      } else {

        const dx = c.cx - parent.cx;

        const sign = dx > 0 ? 1 : -1;

        line(`M ${parent.cx} ${parent.bot}

              L ${parent.cx} ${midY - R}

              Q ${parent.cx} ${midY} ${parent.cx + sign*R} ${midY}

              L ${c.cx - sign*R} ${midY}

              Q ${c.cx} ${midY} ${c.cx} ${midY + R}

              L ${c.cx} ${c.top}`);

      }

    } else {

      line(`M ${parent.cx} ${parent.bot} L ${parent.cx} ${midY}`);

      const leftCX  = Math.min(...children.map(c => c.cx));

      const rightCX = Math.max(...children.map(c => c.cx));

      line(`M ${leftCX} ${midY} L ${rightCX} ${midY}`);

      for (const c of children) {

        line(`M ${c.cx} ${midY} L ${c.cx} ${c.top}`);

      }

    }

  }


  for (const [fromId, toId] of dottedExtra) {

    const from = mid(fromId);

    const to   = mid(toId);

    if (!from || !to) continue;

    const el = document.createElementNS('http://www.w3.org/2000/svg', 'path');

    el.setAttribute('d', `M ${from.cx} ${from.bot} C ${from.cx} ${from.bot+60} ${to.cx} ${to.bot+60} ${to.cx} ${to.bot}`);

    el.setAttribute('stroke', '#c9a84c');

    el.setAttribute('stroke-width', '1.6');

    el.setAttribute('stroke-dasharray', '5 4');

    el.setAttribute('fill', 'none');

    el.setAttribute('stroke-linecap', 'round');

    svg.appendChild(el);

  }

}


window.addEventListener('load', () => { setTimeout(draw, 120); });

window.addEventListener('resize', () => { setTimeout(draw, 80); });

</script>
